# Phase 2: SEO Module — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 8 SEO tools to the Marketolog MCP server — technical audit, AI-SEO check, keyword research (Wordstat API), position tracking (Yandex Search API), competitor analysis, content gap, keyword clustering, and Yandex Webmaster reports.

**Architecture:** Each tool is a standalone async function in its own file under `src/marketolog/modules/seo/`. Tools receive `config` (API keys) and `cache` (FileCache) via closure from server registration. All external API calls go through `fetch_with_retry()`. Results use `format_tabular()` for CSV where appropriate. Graceful degradation: if an API key is missing, tools return setup instructions instead of failing.

**Tech Stack:** Python 3.11+, FastMCP 3.0+, httpx (via fetch_with_retry), beautifulsoup4+lxml (HTML parsing), pyyaml, existing utils (FileCache, format_tabular, fetch_with_retry)

**Spec:** `docs/superpowers/specs/2026-04-07-marketolog-design.md` — Section 6 (SEO Module)

**Existing code interfaces:**
- `MarketologConfig` — dataclass with `.yandex_oauth_token`, `.yandex_wordstat_token`, `.yandex_search_api_key`, `.yandex_folder_id`, `.pagespeed_api_key`, `.exa_api_key`, `.is_configured(field_name)`
- `ProjectContext` — `.get_context()` returns dict with `url`, `seo.main_keywords`, `competitors`, etc.
- `FileCache` — `.get(namespace, key)`, `.set(namespace, key, data, *, ttl_seconds=)`
- `fetch_with_retry(url, *, method, headers, json, params)` → `httpx.Response`
- `format_tabular(data: list[dict])` → CSV string
- `server.py` — `create_server(base_dir)`, tools registered via `@mcp.tool(annotations=...)`, uses `READ_ONLY`/`MUTATING`/`DESTRUCTIVE` constants

```python
# src/marketolog/modules/seo/audit.py
"""Technical SEO audit — PageSpeed API + HTML parsing."""

from urllib.parse import urljoin

from bs4 import BeautifulSoup

from marketolog.core.config import MarketologConfig
from marketolog.utils.cache import FileCache
from marketolog.utils.http import fetch_with_retry

PAGESPEED_URL = "https://pagespeedonline.googleapis.com/pagespeedonline/v5/runPagespeed"

# TTL: 1 hour (technical issues may be fixed quickly)
CACHE_TTL = 3600


async def _fetch_pagespeed(url: str, config: MarketologConfig) -> dict:
    """Fetch Core Web Vitals from PageSpeed Insights API."""
    params = {"url": url, "strategy": "mobile", "category": "performance"}
    if config.pagespeed_api_key:
        params["key"] = config.pagespeed_api_key

    resp = await fetch_with_retry(PAGESPEED_URL, params=params, timeout=60.0)
    if resp.status_code != 200:
        return {"error": f"PageSpeed API: {resp.status_code}"}
    return resp.json()


async def _fetch_page(url: str) -> tuple[str, int]:
    """Fetch HTML content of a page."""
    resp = await fetch_with_retry(url, timeout=30.0)
    return resp.text, resp.status_code


async def _check_robots_txt(base_url: str) -> dict:
    """Check robots.txt existence and contents."""
    url = urljoin(base_url, "/robots.txt")
    resp = await fetch_with_retry(url, timeout=15.0)
    if resp.status_code != 200:
        return {"exists": False, "content": ""}
    return {"exists": True, "content": resp.text}


async def _check_sitemap(base_url: str, robots_content: str) -> dict:
    """Check sitemap existence (from robots.txt or default location)."""
    sitemap_url = None
    for line in robots_content.split("\n"):
        if line.lower().startswith("sitemap:"):
            sitemap_url = line.split(":", 1)[1].strip()
            break

    if not sitemap_url:
        sitemap_url = urljoin(base_url, "/sitemap.xml")

    resp = await fetch_with_retry(sitemap_url, timeout=15.0)
    return {"exists": resp.status_code == 200, "url": sitemap_url}


def _analyze_html(html: str, url: str) -> dict:
    """Parse HTML and extract SEO-relevant data."""
    soup = BeautifulSoup(html, "lxml")

    # Title
    title_tag = soup.find("title")
    title = title_tag.get_text(strip=True) if title_tag else None

    # Meta description
    meta_desc = soup.find("meta", attrs={"name": "description"})
    description = meta_desc["content"] if meta_desc and meta_desc.get("content") else None

    # H1
    h1_tags = soup.find_all("h1")
    h1s = [h.get_text(strip=True) for h in h1_tags]

    # H2
    h2_tags = soup.find_all("h2")
    h2s = [h.get_text(strip=True) for h in h2_tags]

    # Canonical
    canonical_tag = soup.find("link", attrs={"rel": "canonical"})
    canonical = canonical_tag["href"] if canonical_tag and canonical_tag.get("href") else None

    # Schema markup (JSON-LD)
    json_ld_tags = soup.find_all("script", attrs={"type": "application/ld+json"})
    has_schema = len(json_ld_tags) > 0

    # Images without alt
    images = soup.find_all("img")
    imgs_no_alt = [img.get("src", "?") for img in images if not img.get("alt")]

    # Meta robots
    meta_robots = soup.find("meta", attrs={"name": "robots"})
    robots_content = meta_robots["content"] if meta_robots and meta_robots.get("content") else "не указан"

    return {
        "title": title,
        "title_length": len(title) if title else 0,
        "description": description,
        "description_length": len(description) if description else 0,
        "h1": h1s,
        "h2": h2s,
        "canonical": canonical,
        "has_schema_markup": has_schema,
        "schema_count": len(json_ld_tags),
        "images_without_alt": imgs_no_alt,
        "meta_robots": robots_content,
    }


def _format_cwv(pagespeed_data: dict) -> str:
    """Format Core Web Vitals from PageSpeed response."""
    if "error" in pagespeed_data:
        return f"PageSpeed: {pagespeed_data['error']}"

    lhr = pagespeed_data.get("lighthouseResult", {})
    audits = lhr.get("audits", {})
    perf_score = lhr.get("categories", {}).get("performance", {}).get("score")

    lcp = audits.get("largest-contentful-paint", {}).get("numericValue", 0) / 1000
    cls = audits.get("cumulative-layout-shift", {}).get("numericValue", 0)
    tbt = audits.get("total-blocking-time", {}).get("numericValue", 0)
    fcp = audits.get("first-contentful-paint", {}).get("numericValue", 0) / 1000

    score_pct = int(perf_score * 100) if perf_score else "?"

    lines = [
        f"Performance Score: {score_pct}/100",
        f"LCP: {lcp:.1f}s {'✓' if lcp < 2.5 else '✗ (>2.5s)'}",
        f"CLS: {cls:.3f} {'✓' if cls < 0.1 else '✗ (>0.1)'}",
        f"TBT: {tbt:.0f}ms {'✓' if tbt < 200 else '✗ (>200ms)'}",
        f"FCP: {fcp:.1f}s {'✓' if fcp < 1.8 else '✗ (>1.8s)'}",
    ]
    return "\n".join(lines)


async def run_seo_audit(
    url: str,
    *,
    config: MarketologConfig,
    cache: FileCache,
) -> str:
    """Run a full technical SEO audit on a URL.

    Combines: PageSpeed (Core Web Vitals), HTML analysis (meta, headings,
    schema, canonical), robots.txt, sitemap check.
    """
    # Check cache
    cached = cache.get("seo_audit", url)
    if cached:
        return f"[Из кэша]\n{cached}"

    from urllib.parse import urlparse
    parsed = urlparse(url)
    base_url = f"{parsed.scheme}://{parsed.netloc}"

    # Parallel-ish fetches (sequential for simplicity, could be asyncio.gather)
    pagespeed_data = await _fetch_pagespeed(url, config)
    html, status_code = await _fetch_page(url)
    robots = await _check_robots_txt(base_url)
    sitemap = await _check_sitemap(base_url, robots.get("content", ""))

    # Analyze HTML
    html_data = _analyze_html(html, url)

    # Format report
    sections = []

    sections.append(f"# SEO-аудит: {url}\n")

    # Core Web Vitals
    sections.append("## Core Web Vitals (mobile)")
    sections.append(_format_cwv(pagespeed_data))

    # Meta tags
    sections.append("\n## Мета-теги")
    t = html_data["title"]
    tl = html_data["title_length"]
    sections.append(f"Title: {t} ({tl} символов) {'✓' if 30 <= tl <= 60 else '⚠ рекомендуется 30-60'}")
    d = html_data["description"]
    dl = html_data["description_length"]
    sections.append(f"Description: {d} ({dl} символов) {'✓' if 70 <= dl <= 160 else '⚠ рекомендуется 70-160'}")
    sections.append(f"Meta robots: {html_data['meta_robots']}")

    # Headings
    sections.append("\n## Заголовки")
    sections.append(f"H1: {html_data['h1'] if html_data['h1'] else '⚠ Нет H1!'}")
    if len(html_data["h1"]) > 1:
        sections.append(f"⚠ Множественные H1: {len(html_data['h1'])}")
    sections.append(f"H2: {len(html_data['h2'])} шт.")

    # Technical
    sections.append("\n## Техническое")
    sections.append(f"Canonical: {html_data['canonical'] or '⚠ Не указан'}")
    sections.append(f"Schema markup (JSON-LD): {'✓' if html_data['has_schema_markup'] else '✗ Нет'} ({html_data['schema_count']} шт.)")
    sections.append(f"Robots.txt: {'✓' if robots['exists'] else '✗ Не найден'}")
    sections.append(f"Sitemap: {'✓' if sitemap['exists'] else '✗ Не найден'} ({sitemap['url']})")

    # Images
    if html_data["images_without_alt"]:
        sections.append(f"\n## Изображения без alt ({len(html_data['images_without_alt'])})")
        for src in html_data["images_without_alt"][:5]:
            sections.append(f"  - {src}")

    report = "\n".join(sections)

    # Cache result
    cache.set("seo_audit", url, report, ttl_seconds=CACHE_TTL)

    return report
```

```python
# src/marketolog/modules/seo/ai_seo.py
"""AI-SEO readiness check — AI crawler access, llms.txt, schema markup."""

from urllib.parse import urljoin, urlparse

from bs4 import BeautifulSoup

from marketolog.utils.cache import FileCache
from marketolog.utils.http import fetch_with_retry

AI_CRAWLERS = ["GPTBot", "ClaudeBot", "PerplexityBot", "GoogleOther", "Applebot-Extended"]
CACHE_TTL = 3600


def _parse_robots_for_ai(robots_text: str) -> dict[str, str]:
    """Parse robots.txt and extract rules for AI crawlers.

    Returns: {crawler_name: "allowed" | "blocked" | "partially blocked" | "not mentioned"}
    """
    results = {c: "не упомянут" for c in AI_CRAWLERS}

    current_agent = None
    for line in robots_text.split("\n"):
        line = line.strip()
        if line.lower().startswith("user-agent:"):
            current_agent = line.split(":", 1)[1].strip()
        elif current_agent in AI_CRAWLERS:
            if line.lower().startswith("disallow: /") and line.strip() == "Disallow: /":
                results[current_agent] = "заблокирован"
            elif line.lower().startswith("disallow:") and len(line.split(":", 1)[1].strip()) > 1:
                results[current_agent] = "частично заблокирован"
            elif line.lower().startswith("allow:"):
                if results[current_agent] == "не упомянут":
                    results[current_agent] = "разрешён"

    return results


async def run_ai_seo_check(
    url: str,
    *,
    cache: FileCache,
) -> str:
    """Check site readiness for AI search engines.

    Checks: AI crawler rules in robots.txt, llms.txt presence,
    schema markup, content accessibility without JS.
    """
    cached = cache.get("ai_seo", url)
    if cached:
        return f"[Из кэша]\n{cached}"

    parsed = urlparse(url)
    base_url = f"{parsed.scheme}://{parsed.netloc}"

    # Fetch robots.txt
    robots_resp = await fetch_with_retry(urljoin(base_url, "/robots.txt"), timeout=15.0)
    robots_text = robots_resp.text if robots_resp.status_code == 200 else ""

    # Check llms.txt
    llms_resp = await fetch_with_retry(urljoin(base_url, "/llms.txt"), timeout=15.0)
    has_llms_txt = llms_resp.status_code == 200

    # Fetch page HTML
    page_resp = await fetch_with_retry(url, timeout=30.0)
    html = page_resp.text if page_resp.status_code == 200 else ""

    # Analyze
    ai_rules = _parse_robots_for_ai(robots_text)

    soup = BeautifulSoup(html, "lxml") if html else None
    has_schema = bool(soup and soup.find_all("script", attrs={"type": "application/ld+json"}))
    has_noscript_content = bool(soup and soup.find("body") and len(soup.find("body").get_text(strip=True)) > 50)

    # Format report
    sections = [f"# AI-SEO проверка: {url}\n"]

    sections.append("## AI-краулеры в robots.txt")
    for crawler, status in ai_rules.items():
        icon = "✓" if status in ("разрешён", "не упомянут") else "⚠" if "частично" in status else "✗"
        sections.append(f"  {icon} {crawler}: {status}")

    sections.append(f"\n## llms.txt")
    if has_llms_txt:
        preview = llms_resp.text[:200]
        sections.append(f"✓ Найден ({len(llms_resp.text)} символов)")
        sections.append(f"Превью: {preview}...")
    else:
        sections.append("✗ Не найден — рекомендуется создать для улучшения видимости в AI-ответах")

    sections.append(f"\n## Schema Markup (JSON-LD)")
    sections.append(f"{'✓' if has_schema else '✗'} {'Найден' if has_schema else 'Не найден'}")

    sections.append(f"\n## Контент без JavaScript")
    sections.append(f"{'✓' if has_noscript_content else '⚠'} {'Доступен' if has_noscript_content else 'Мало текстового контента без JS'}")

    # Recommendations
    recs = []
    blocked_crawlers = [c for c, s in ai_rules.items() if s == "заблокирован"]
    if blocked_crawlers:
        recs.append(f"Разблокировать AI-краулеры: {', '.join(blocked_crawlers)} (если нет причин блокировать)")
    if not has_llms_txt:
        recs.append("Создать llms.txt — описание сайта для AI-поисковиков (https://llmstxt.org)")
    if not has_schema:
        recs.append("Добавить JSON-LD schema markup (Organization, Product, FAQ)")

    if recs:
        sections.append("\n## Рекомендации")
        for i, rec in enumerate(recs, 1):
            sections.append(f"{i}. {rec}")

    report = "\n".join(sections)
    cache.set("ai_seo", url, report, ttl_seconds=CACHE_TTL)
    return report
```

```python
"""Keyword research via Yandex Wordstat API + keyword clustering."""

from collections import defaultdict

from marketolog.core.config import MarketologConfig
from marketolog.utils.cache import FileCache
from marketolog.utils.formatting import format_tabular
from marketolog.utils.http import fetch_with_retry

WORDSTAT_BASE = "https://api.wordstat.yandex.net"
CACHE_TTL = 86400  # 24 hours


async def _wordstat_top_requests(keyword: str, config: MarketologConfig) -> list[dict]:
    """Call Wordstat API /v1/topRequests."""
    resp = await fetch_with_retry(
        f"{WORDSTAT_BASE}/v1/topRequests",
        method="POST",
        headers={"Authorization": f"Bearer {config.yandex_wordstat_token}"},
        json={"text": keyword, "top": 50},
    )
    if resp.status_code != 200:
        return []
    data = resp.json()
    return data.get("result", [])


async def run_keyword_research(
    seed_keywords: list[str],
    *,
    config: MarketologConfig,
    cache: FileCache,
    count: int = 50,
) -> str:
    """Research keywords via Yandex Wordstat API.

    Args:
        seed_keywords: Starting keywords to expand.
        config: API credentials.
        cache: File cache.
        count: Max results per seed keyword.

    Returns:
        Formatted keyword list with volumes (CSV format).
    """
    if not config.is_configured("yandex_wordstat_token"):
        return (
            "⚠ Яндекс Wordstat API не настроен.\n\n"
            "Для работы keyword_research нужен токен:\n"
            "1. Создайте приложение на oauth.yandex.ru\n"
            "2. Подайте заявку на доступ к Wordstat API (~24 часа)\n"
            "3. Получите OAuth-токен\n"
            "4. Установите: python -m marketolog auth wordstat\n"
            "   или YANDEX_WORDSTAT_TOKEN в env"
        )

    all_keywords = []

    for seed in seed_keywords:
        # Check cache
        cached = cache.get("wordstat", seed)
        if cached:
            all_keywords.extend(cached)
            continue

        results = await _wordstat_top_requests(seed, config)
        if results:
            cache.set("wordstat", seed, results, ttl_seconds=CACHE_TTL)
            all_keywords.extend(results)

    if not all_keywords:
        return "Не найдено ключевых слов. Попробуйте другие seed-запросы."

    # Deduplicate by text
    seen = set()
    unique = []
    for kw in all_keywords:
        text = kw.get("text", "")
        if text not in seen:
            seen.add(text)
            unique.append(kw)

    # Sort by count descending
    unique.sort(key=lambda x: x.get("count", 0), reverse=True)
    unique = unique[:count]

    return format_tabular(unique)


def run_keyword_cluster(keywords: list[dict]) -> list[dict]:
    """Cluster keywords by common stems/words.

    Simple approach: group keywords that share 2+ common words.
    """
    if not keywords:
        return []

    # Extract word sets for each keyword
    kw_words = []
    for kw in keywords:
        text = kw.get("text", "")
        words = set(text.lower().split())
        kw_words.append((kw, words))

    # Group by largest common word overlap
    clusters: dict[str, list[dict]] = defaultdict(list)
    used = set()

    # Sort by count so high-volume keywords anchor clusters
    sorted_kws = sorted(kw_words, key=lambda x: x[0].get("count", 0), reverse=True)

    for i, (kw, words) in enumerate(sorted_kws):
        if i in used:
            continue

        # This keyword starts a cluster
        cluster_name = kw["text"]
        cluster_kws = [kw]
        used.add(i)

        for j, (other_kw, other_words) in enumerate(sorted_kws):
            if j in used:
                continue
            # At least 1 shared meaningful word (>2 chars)
            shared = {w for w in words & other_words if len(w) > 2}
            if shared:
                cluster_kws.append(other_kw)
                used.add(j)

        clusters[cluster_name] = cluster_kws

    return [
        {"name": name, "keywords": kws, "total_volume": sum(k.get("count", 0) for k in kws)}
        for name, kws in clusters.items()
    ]
```

```python
# src/marketolog/modules/seo/positions.py
"""Position tracking via Yandex Search API v2 (XML)."""

from urllib.parse import urlparse
from xml.etree import ElementTree as ET

from marketolog.core.config import MarketologConfig
from marketolog.utils.cache import FileCache
from marketolog.utils.formatting import format_tabular
from marketolog.utils.http import fetch_with_retry

YANDEX_SEARCH_URL = "https://yandex.ru/search/xml"
CACHE_TTL = 21600  # 6 hours


async def _search_yandex(query: str, config: MarketologConfig) -> list[dict]:
    """Search Yandex and return list of {url, title, position}."""
    resp = await fetch_with_retry(
        YANDEX_SEARCH_URL,
        params={
            "user": config.yandex_search_api_key,
            "key": config.yandex_folder_id,
            "query": query,
            "lr": "213",  # Moscow region
            "groupby": "attr=d.mode=deep.groups-on-page=50.docs-in-group=1",
        },
        timeout=30.0,
    )
    if resp.status_code != 200:
        return []

    results = []
    try:
        root = ET.fromstring(resp.text)
        groups = root.findall(".//group")
        for i, group in enumerate(groups, 1):
            doc = group.find("doc")
            if doc is not None:
                url_el = doc.find("url")
                title_el = doc.find("title")
                results.append({
                    "position": i,
                    "url": url_el.text if url_el is not None else "",
                    "title": title_el.text if title_el is not None else "",
                })
    except ET.ParseError:
        pass

    return results


async def run_check_positions(
    keywords: list[str],
    site_url: str,
    *,
    config: MarketologConfig,
    cache: FileCache,
) -> str:
    """Check site positions in Yandex search for given keywords.

    Args:
        keywords: Keywords to check.
        site_url: The site URL to find in results.
        config: API credentials.
        cache: File cache.

    Returns:
        CSV table: keyword, position, url, title
    """
    if not config.is_configured("yandex_search_api_key"):
        return (
            "⚠ Яндекс Поиск API не настроен.\n\n"
            "Для проверки позиций нужен API-ключ:\n"
            "1. Создайте аккаунт в Yandex Cloud\n"
            "2. Получите API-ключ для Yandex Search API v2\n"
            "3. Установите YANDEX_SEARCH_API_KEY и YANDEX_FOLDER_ID в env"
        )

    parsed_site = urlparse(site_url)
    site_domain = parsed_site.netloc

    position_data = []

    for keyword in keywords:
        # Check cache
        cache_key = f"{keyword}|{site_domain}"
        cached = cache.get("positions", cache_key)
        if cached:
            position_data.append(cached)
            continue

        results = await _search_yandex(keyword, config)

        # Find site in results
        found = False
        for r in results:
            result_domain = urlparse(r["url"]).netloc
            if site_domain in result_domain or result_domain in site_domain:
                entry = {
                    "keyword": keyword,
                    "position": r["position"],
                    "url": r["url"],
                    "title": r["title"],
                }
                position_data.append(entry)
                cache.set("positions", cache_key, entry, ttl_seconds=CACHE_TTL)
                found = True
                break

        if not found:
            entry = {
                "keyword": keyword,
                "position": "не найден (>50)",
                "url": "—",
                "title": "—",
            }
            position_data.append(entry)

    if not position_data:
        return "Нет данных о позициях."

    return format_tabular(position_data)
```

```python
"""Yandex Webmaster API — indexing, errors, popular queries."""

from marketolog.core.config import MarketologConfig
from marketolog.utils.cache import FileCache
from marketolog.utils.formatting import format_tabular
from marketolog.utils.http import fetch_with_retry

WEBMASTER_BASE = "https://api.webmaster.yandex.net/v4"
CACHE_TTL = 3600


def _auth_headers(config: MarketologConfig) -> dict[str, str]:
    return {"Authorization": f"OAuth {config.yandex_oauth_token}"}


async def _get_user_id(config: MarketologConfig) -> str | None:
    """Get Yandex Webmaster user ID."""
    resp = await fetch_with_retry(
        f"{WEBMASTER_BASE}/user",
        headers=_auth_headers(config),
    )
    if resp.status_code != 200:
        return None
    return str(resp.json().get("user_id", ""))


async def _get_host_id(user_id: str, host: str, config: MarketologConfig) -> str | None:
    """Find host_id for given host URL."""
    resp = await fetch_with_retry(
        f"{WEBMASTER_BASE}/user/{user_id}/hosts",
        headers=_auth_headers(config),
    )
    if resp.status_code != 200:
        return None

    for h in resp.json().get("hosts", []):
        if host in h.get("ascii_host_url", ""):
            return h["host_id"]
    return None


async def _get_popular_queries(user_id: str, host_id: str, config: MarketologConfig) -> list[dict]:
    """Get popular search queries."""
    resp = await fetch_with_retry(
        f"{WEBMASTER_BASE}/user/{user_id}/hosts/{host_id}/search-queries/popular",
        headers=_auth_headers(config),
    )
    if resp.status_code != 200:
        return []
    return resp.json().get("queries", [])


async def _get_diagnostics(user_id: str, host_id: str, config: MarketologConfig) -> dict:
    """Get indexing diagnostics."""
    resp = await fetch_with_retry(
        f"{WEBMASTER_BASE}/user/{user_id}/hosts/{host_id}/diagnostics",
        headers=_auth_headers(config),
    )
    if resp.status_code != 200:

async def run_webmaster_report(
    host: str,
    *,
    config: MarketologConfig,
    cache: FileCache,

    """Generate Yandex Webmaster report: indexing, errors, popular queries."""
    if not config.is_configured("yandex_oauth_token"):

            "⚠ Яндекс Вебмастер не настроен.\n\n"

            "1. Создайте приложение на oauth.yandex.ru (scopes: webmaster:*)\n"
            "2. Получите токен: python -m marketolog auth yandex\n"
            "   или YANDEX_OAUTH_TOKEN в env"
        )

    cached = cache.get("webmaster", host)
    if cached:
        return f"[Из кэша]\n{cached}"

    # Get user and host IDs
    user_id = await _get_user_id(config)
    if not user_id:
        return "Ошибка: не удалось получить user_id Вебмастера."

    host_id = await _get_host_id(user_id, host, config)
    if not host_id:
        return f"Сайт {host} не найден в Яндекс.Вебмастере. Добавьте его на webmaster.yandex.ru."

    # Fetch data
    queries = await _get_popular_queries(user_id, host_id, config)
    diagnostics = await _get_diagnostics(user_id, host_id, config)
    indexing = await _get_indexing_history(user_id, host_id, config)

    # Format report
    sections = [f"# Яндекс.Вебмастер: {host}\n"]

    # Indexing
    sections.append("## Индексация")
    searchable = indexing.get("indicators", {}).get("SEARCHABLE", {})
    values = searchable.get("values", [])
    if values:
        sections.append(f"Страниц в поиске: {values[-1]} (последнее значение)")
        if len(values) >= 2:
            diff = values[-1] - values[0]
            sections.append(f"Динамика: {'+' if diff >= 0 else ''}{diff} за период")

    # Diagnostics
    sections.append("\n## Проблемы")
    problems = diagnostics.get("problems", [])
    if problems:
        for p in problems:
            icon = "🔴" if p.get("severity") == "CRITICAL" else "🟡"
            sections.append(f"  {icon} {p.get('type', '?')}: {p.get('count', '?')} шт. ({p.get('severity', '?')})")
    else:
        sections.append("  Проблем не обнаружено")

    # Popular queries
    sections.append("\n## Популярные запросы")
    if queries:
        table_data = [
            {
                "запрос": q.get("query_text", ""),
                "показы": q.get("impressions", 0),
                "клики": q.get("clicks", 0),
                "позиция": round(q.get("position", 0), 1),
            }
            for q in queries[:20]
        ]
        sections.append(format_tabular(table_data))
    else:
        sections.append("  Нет данных о запросах")

    report = "\n".join(sections)
    cache.set("webmaster", host, report, ttl_seconds=CACHE_TTL)
    return report
```

```python
# src/marketolog/modules/seo/competitors.py
"""Competitor analysis and content gap identification."""

from urllib.parse import urlparse

from bs4 import BeautifulSoup

from marketolog.core.config import MarketologConfig
from marketolog.utils.cache import FileCache
from marketolog.utils.formatting import format_tabular
from marketolog.utils.http import fetch_with_retry
from marketolog.modules.seo.positions import _search_yandex

CACHE_TTL = 3600


async def _analyze_single_competitor(url: str) -> dict:
    """Fetch and analyze a competitor's page."""
    resp = await fetch_with_retry(url, timeout=30.0)
    if resp.status_code != 200:
        return {"url": url, "error": f"HTTP {resp.status_code}"}

    soup = BeautifulSoup(resp.text, "lxml")

    title_tag = soup.find("title")
    title = title_tag.get_text(strip=True) if title_tag else "—"

    meta_desc = soup.find("meta", attrs={"name": "description"})
    description = meta_desc["content"] if meta_desc and meta_desc.get("content") else "—"

    h1_tags = soup.find_all("h1")
    h2_tags = soup.find_all("h2")

    # Count content sections
    text_length = len(soup.get_text(strip=True))

    # Check for schema
    schema_tags = soup.find_all("script", attrs={"type": "application/ld+json"})

    return {
        "url": url,
        "title": title,
        "description": description,
        "h1": [h.get_text(strip=True) for h in h1_tags],
        "h2_count": len(h2_tags),
        "h2_topics": [h.get_text(strip=True) for h in h2_tags[:10]],
        "text_length": text_length,
        "has_schema": len(schema_tags) > 0,
    }


async def run_analyze_competitors(
    competitor_urls: list[str],
    *,
    config: MarketologConfig,
    cache: FileCache,
) -> str:
    """Analyze competitor websites: structure, content, meta tags."""
    sections = ["# Анализ конкурентов\n"]

    for url in competitor_urls:
        cached = cache.get("competitors", url)
        if cached:
            sections.append(cached)
            continue

        data = await _analyze_single_competitor(url)

        if "error" in data:
            report = f"## {url}\n⚠ Ошибка: {data['error']}"
        else:
            report_lines = [
                f"## {url}",
                f"Title: {data['title']}",
                f"Description: {data['description']}",
                f"H1: {data['h1']}",
                f"Секций (H2): {data['h2_count']}",
                f"Темы: {', '.join(data['h2_topics'])}",
                f"Объём текста: ~{data['text_length']} символов",
                f"Schema markup: {'✓' if data['has_schema'] else '✗'}",
            ]
            report = "\n".join(report_lines)

        cache.set("competitors", url, report, ttl_seconds=CACHE_TTL)
        sections.append(report)

    return "\n\n".join(sections)


async def run_content_gap(
    site_url: str,
    competitor_urls: list[str],
    keywords: list[str],
    *,
    config: MarketologConfig,
    cache: FileCache,
) -> str:
    """Find keywords where competitors rank but you don't.

    Compares Yandex search results for each keyword.
    """
    if not config.is_configured("yandex_search_api_key"):
        return (
            "⚠ Для content_gap нужен Яндекс Поиск API.\n"
            "Установите YANDEX_SEARCH_API_KEY и YANDEX_FOLDER_ID."
        )

    site_domain = urlparse(site_url).netloc
    competitor_domains = {urlparse(u).netloc for u in competitor_urls}

    gaps = []

    for keyword in keywords:
        results = await _search_yandex(keyword, config)

        site_position = None
        competitor_positions = {}

        for r in results:
            domain = urlparse(r["url"]).netloc
            if site_domain in domain:
                site_position = r["position"]
            for cd in competitor_domains:
                if cd in domain:
                    competitor_positions[cd] = r["position"]

        # Gap: competitor ranks, we don't (or rank much worse)
        if competitor_positions and (site_position is None or site_position > 20):
            gaps.append({
                "keyword": keyword,
                "your_position": site_position or ">50",
                "competitor_best": min(competitor_positions.values()),
                "competitor": min(competitor_positions, key=competitor_positions.get),
            })

    if not gaps:
        return "Контент-гэпов не обнаружено — вы покрываете те же запросы, что и конкуренты."

    return "# Content Gap\n\n" + format_tabular(gaps)
```
